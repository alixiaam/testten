Proyecto 2 | Reloj

Explicaion

Este código muestra un reloj digital en formato de 12 horas (AM/PM) en una página web. Primero obtiene la hora actual usando new Date(), luego ajusta las horas para que se muestren correctamente en formato de 12 horas (cambiando a "PM" si es después del mediodía). Si las horas, minutos o segundos son menores a 10, les agrega un "0" delante para que siempre tengan dos dígitos. Luego, actualiza el contenido de un elemento en la página con el ID "MyClockDisplay" para mostrar la hora actual y repite este proceso cada segundo con setTimeout().


            function showTime(){
                var date = new Date();
                var h = date.getHours(); // 0 - 23
                var m = date.getMinutes(); // 0 - 59
                var s = date.getSeconds(); // 0 - 59
                var session = "AM";
                
                if(h == 0){
                    h = 12;
                }
                
                if(h > 12){
                    h = h - 12;
                    session = "PM";
                }
                
                h = (h < 10) ? "0" + h : h;
                m = (m < 10) ? "0" + m : m;
                s = (s < 10) ? "0" + s : s;
                
                var time = h + ":" + m + ":" + s + " " + session;
                document.getElementById("MyClockDisplay").innerText = time;
                document.getElementById("MyClockDisplay").textContent = time;
                
                setTimeout(showTime, 1000);
                
            }
            
            showTime();
